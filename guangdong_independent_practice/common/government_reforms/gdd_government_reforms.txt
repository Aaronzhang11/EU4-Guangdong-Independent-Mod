gdd_local_fiefdom_reform = {
    icon = "gdd_imperial_bulwark"
    allow_normal_conversion = no
    lock_level_when_selected = yes
    valid_for_new_country = no
    valid_for_nation_designer = no
    nation_designer_cost = 0

    # Countries can only receive this reform while their capital is in the
    # China superregion. Once granted, it remains valid after the capital is
    # moved abroad until the dedicated "Leave China" event removes it.
    potential = {
        OR = {
            has_reform = gdd_local_fiefdom_reform
            capital_scope = {
                superregion = china_superregion
            }
        }
    }

    modifiers = {
        global_tax_modifier = 0.30
        num_accepted_cultures = 1
        diplomats = 1
    }

    ai = {
        factor = 0
    }
}
