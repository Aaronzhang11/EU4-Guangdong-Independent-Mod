gdd_nanhai_temple = {
	start = 667 # Guangzhou
	date = 594.1.1 # Traditionally founded under Emperor Wen of Sui
	time = { months = 0 }
	build_cost = 0
	can_be_moved = no
	move_days_per_unit_distance = 10
	starting_tier = 1
	type = monument

	build_trigger = {
	}

	on_built = {
	}

	on_destroyed = {
	}

	can_use_modifiers_trigger = {
		custom_trigger_tooltip = {
			tooltip = gdd_nanhai_temple_culture_core_tt
			is_core = owner
			owner = {
				OR = {
					primary_culture = cantonese
					accepted_culture = cantonese
				}
			}
		}
	}

	can_upgrade_trigger = {
		custom_trigger_tooltip = {
			tooltip = gdd_nanhai_temple_culture_core_tt
			is_core = owner
			owner = {
				OR = {
					primary_culture = cantonese
					accepted_culture = cantonese
				}
			}
		}

		custom_trigger_tooltip = {
			tooltip = gdd_nanhai_temple_controlled_tt
			controlled_by = owner
		}

		# Tier 1 -> Tier 2: restore the official state sacrifices in stable times.
		if = {
			limit = {
				has_great_project = {
					type = gdd_nanhai_temple
					tier = 1
				}
				NOT = {
					has_great_project = {
						type = gdd_nanhai_temple
						tier = 2
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = gdd_nanhai_temple_tier_2_upgrade_tt
				owner = {
					stability = 1
				}
			}
		}

		# Tier 2 -> Tier 3: connect Fuxu harbor to a true South Seas network.
		if = {
			limit = {
				has_great_project = {
					type = gdd_nanhai_temple
					tier = 2
				}
				NOT = {
					has_great_project = {
						type = gdd_nanhai_temple
						tier = 3
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = gdd_nanhai_temple_tier_3_upgrade_tt
				province_has_center_of_trade_of_level = 3
				owner = {
					num_of_owned_provinces_with = {
						value = 3
						has_port = yes
						is_core = owner
						OR = {
							region = malaya_region
							region = indonesia_region
						}
					}
				}
			}
		}
	}

	# The temple survives conquest, but an owner must meet the culture and core
	# requirements above before receiving its modifiers or upgrading it.
	keep_trigger = {
	}

	tier_0 = {
		upgrade_time = { months = 0 }
		cost_to_upgrade = { factor = 0 }
		province_modifiers = {
		}
		area_modifier = {
		}
		country_modifiers = {
		}
		on_upgraded = {
		}
	}

	tier_1 = {
		upgrade_time = { months = 120 }
		cost_to_upgrade = { factor = 1000 }
		province_modifiers = {
			province_trade_power_modifier = 0.15
			local_sailors_modifier = 0.25
		}
		area_modifier = {
		}
		country_modifiers = {
			global_ship_trade_power = 0.05
		}
		on_upgraded = {
		}
	}

	tier_2 = {
		upgrade_time = { months = 240 }
		cost_to_upgrade = { factor = 2500 }
		province_modifiers = {
			province_trade_power_modifier = 0.30
			local_sailors_modifier = 0.50
		}
		area_modifier = {
		}
		country_modifiers = {
			trade_steering = 0.05
			global_ship_trade_power = 0.10
			global_sailors_modifier = 0.10
		}
		on_upgraded = {
		}
	}

	tier_3 = {
		upgrade_time = { months = 480 }
		cost_to_upgrade = { factor = 5000 }
		province_modifiers = {
			province_trade_power_modifier = 0.50
			local_sailors_modifier = 0.75
		}
		area_modifier = {
		}
		country_modifiers = {
			trade_steering = 0.10
			global_ship_trade_power = 0.20
			global_sailors_modifier = 0.15
			merchants = 1
		}
		on_upgraded = {
		}
	}
}
