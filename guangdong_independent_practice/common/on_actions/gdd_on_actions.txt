on_startup = {
    events = {
        gdd_startup.1
        gdd_heir.1
        gdd_subject.108
    }

    # Grant Guangdong's two founding estate privileges exactly once per campaign.
    # The flag prevents an invalidated privilege from returning after a later reload.
    if = {
        limit = {
            tag = GDD
            NOT = { has_country_flag = gdd_estate_privileges_initialised }
        }
        set_country_flag = gdd_estate_privileges_initialised
        set_estate_privilege = estate_nobles_gdd_clan_autonomy_charter
        if = {
            limit = { num_of_ports = 1 }
            set_estate_privilege = estate_burghers_gdd_southern_trade_charter
        }
    }

    # Convert tax buildings in old saves before the player opens the building
    # interface.  Their statistics and tier remain unchanged.
    if = {
        limit = { tag = GDD }
        every_owned_province = {
            gdd_sync_tax_building_presentation_effect = yes
        }
    }
}

# The scheduled five-year checker handles normal 1444 starts. The treaty check
# is also called monthly for old saves and console date jumps. Tribute is
# checked here in January so its date cannot drift across leap years.
on_monthly_pulse = {
    events = {
        gdd_subject.108
        gdd_treaty.100
        gdd_subject.105
        gdd_subject.106
    }

    # Catch buildings granted by vanilla missions or third-party scripted
    # effects which explicitly add temple/cathedral to Guangdong.
    if = {
        limit = { tag = GDD }
        every_owned_province = {
            gdd_sync_tax_building_presentation_effect = yes
        }
    }
}

on_new_monarch = {
    events = {
        gdd_heir.2
        gdd_startup.6
    }
}

on_heir_death = {
    country_event = { id = gdd_heir.2 }
}

on_heir_disinherited = {
    country_event = { id = gdd_heir.2 }
}

# Province scope.  A Guangdong province immediately receives the ancestral
# hall presentation; a province leaving Guangdong returns to vanilla names and
# icons.
on_province_owner_change = {
    gdd_sync_tax_building_presentation_effect = yes
}
